# PrincipiaBlastFoam 量化指标统计实现方案

## 概述
目前项目已实现完整的自动化工作流，但缺少对关键指标的量化统计。本文档提供三个实现方案，从简单到复杂，供您根据需求选择。

## 需要统计的核心指标

### 1. LLM 相关指标
- **Token 消耗量**
  - 输入 tokens (prompt tokens)
  - 输出 tokens (completion tokens)
  - 总 tokens
  - 各 Agent 的 token 分布
  
### 2. 工作流指标
- **迭代次数**
  - 总迭代次数
  - 各 Agent 执行次数
  - Reviewer 验证次数
  - 失败重试次数
  
### 3. 时间指标
- **执行时间**
  - 总执行时间
  - 各 Agent 耗时
  - LLM 调用耗时
  - 文件操作耗时
  
### 4. 任务指标
- **任务统计**
  - 计划任务数
  - 完成任务数
  - 失败任务数
  - 任务成功率
  
### 5. 质量指标
- **代码质量**
  - 验证通过率
  - 修复次数
  - 错误类型分布

```python
# principia_ai/metrics/tracker.py
import time
from typing import Dict, Any, List
from collections import defaultdict
import json
from datetime import datetime

class MetricsTracker:
    """单例模式的指标追踪器"""
    _instance = None
    
    def __new__(cls):
        if cls._instance is None:
            cls._instance = super().__new__(cls)
            cls._instance._initialize()
        return cls._instance
    
    def _initialize(self):
        self.metrics = {
            'task_id': None,
            'start_time': None,
            'end_time': None,
            'llm_calls': [],  # 每次 LLM 调用的详细信息
            'agent_executions': defaultdict(int),  # 各 agent 执行次数
            'agent_timings': defaultdict(list),  # 各 agent 执行时间
            'total_tokens': {'input': 0, 'output': 0, 'total': 0},
            'tokens_by_agent': defaultdict(lambda: {'input': 0, 'output': 0, 'total': 0}),
            'tasks': {
                'planned': 0,
                'completed': 0,
                'failed': 0,
                'retried': 0
            },
            'validation_results': {
                'passed': 0,
                'failed': 0
            },
            'errors': []
        }
        self.current_agent = None
        self.current_start_time = None
    
    def start_task(self, task_id: str):
        """开始新任务"""
        self.metrics['task_id'] = task_id
        self.metrics['start_time'] = datetime.now().isoformat()
    
    def end_task(self):
        """结束任务"""
        self.metrics['end_time'] = datetime.now().isoformat()
    
    def start_agent(self, agent_name: str):
        """记录 Agent 开始执行"""
        self.current_agent = agent_name
        self.current_start_time = time.time()
        self.metrics['agent_executions'][agent_name] += 1
    
    def end_agent(self, agent_name: str):
        """记录 Agent 结束执行"""
        if self.current_start_time:
            duration = time.time() - self.current_start_time
            self.metrics['agent_timings'][agent_name].append(duration)
        self.current_agent = None
        self.current_start_time = None
    
    def record_llm_call(self, agent_name: str, 
                       input_tokens: int, 
                       output_tokens: int,
                       model: str,
                       duration: float = None):
        """记录 LLM 调用"""
        total = input_tokens + output_tokens
        
        call_info = {
            'agent': agent_name,
            'timestamp': datetime.now().isoformat(),
            'input_tokens': input_tokens,
            'output_tokens': output_tokens,
            'total_tokens': total,
            'model': model,
            'duration': duration
        }
        self.metrics['llm_calls'].append(call_info)
        
        # 更新总计
        self.metrics['total_tokens']['input'] += input_tokens
        self.metrics['total_tokens']['output'] += output_tokens
        self.metrics['total_tokens']['total'] += total
        
        # 按 Agent 统计
        self.metrics['tokens_by_agent'][agent_name]['input'] += input_tokens
        self.metrics['tokens_by_agent'][agent_name]['output'] += output_tokens
        self.metrics['tokens_by_agent'][agent_name]['total'] += total
    
    def record_task_event(self, event_type: str):
        """记录任务事件"""
        if event_type in self.metrics['tasks']:
            self.metrics['tasks'][event_type] += 1
    
    def record_validation(self, passed: bool):
        """记录验证结果"""
        if passed:
            self.metrics['validation_results']['passed'] += 1
        else:
            self.metrics['validation_results']['failed'] += 1
    
    def record_error(self, agent_name: str, error_msg: str):
        """记录错误"""
        self.metrics['errors'].append({
            'agent': agent_name,
            'timestamp': datetime.now().isoformat(),
            'error': error_msg
        })
    
    def get_metrics(self) -> Dict[str, Any]:
        """获取当前指标"""
        return dict(self.metrics)
    
    def reset(self):
        """重置指标"""
        self._initialize()
```

```python
# principia_ai/metrics/decorators.py
import time
import functools
from typing import Callable
from .tracker import MetricsTracker

def track_agent_execution(agent_name: str):
    """装饰器：追踪 Agent 执行"""
    def decorator(func: Callable):
        @functools.wraps(func)
        def wrapper(*args, **kwargs):
            tracker = MetricsTracker()
            tracker.start_agent(agent_name)
            try:
                result = func(*args, **kwargs)
                return result
            except Exception as e:
                tracker.record_error(agent_name, str(e))
                raise
            finally:
                tracker.end_agent(agent_name)
        return wrapper
    return decorator

def track_llm_call(agent_name: str):
    """装饰器：追踪 LLM 调用"""
    def decorator(func: Callable):
        @functools.wraps(func)
        def wrapper(self, *args, **kwargs):
            tracker = MetricsTracker()
            start_time = time.time()
            
            # 执行原函数
            response = func(self, *args, **kwargs)
            
            duration = time.time() - start_time
            
            # 提取 token 使用信息
            if hasattr(response, 'usage_metadata'):
                usage = response.usage_metadata
                tracker.record_llm_call(
                    agent_name=agent_name,
                    input_tokens=getattr(usage, 'input_tokens', 0),
                    output_tokens=getattr(usage, 'output_tokens', 0),
                    model=getattr(self.llm, 'model_name', 'unknown'),
                    duration=duration
                )
            elif hasattr(response, 'response_metadata'):
                # 兼容不同的响应格式
                metadata = response.response_metadata
                if 'token_usage' in metadata:
                    usage = metadata['token_usage']
                    tracker.record_llm_call(
                        agent_name=agent_name,
                        input_tokens=usage.get('prompt_tokens', 0),
                        output_tokens=usage.get('completion_tokens', 0),
                        model=metadata.get('model_name', 'unknown'),
                        duration=duration
                    )
            
            return response
        return wrapper
    return decorator
```

```python
# principia_ai/metrics/reporter.py
import json
from datetime import datetime
from typing import Dict, Any
from .tracker import MetricsTracker

class MetricsReporter:
    """生成指标报告"""
    
    @staticmethod
    def generate_summary(metrics: Dict[str, Any]) -> str:
        """生成文本摘要"""
        lines = []
        lines.append("=" * 60)
        lines.append("  PrincipiaBlastFoam 工作流执行报告")
        lines.append("=" * 60)
        lines.append(f"任务ID: {metrics['task_id']}")
        lines.append(f"开始时间: {metrics['start_time']}")
        lines.append(f"结束时间: {metrics['end_time']}")
        
        # 计算总时间
        if metrics['start_time'] and metrics['end_time']:
            start = datetime.fromisoformat(metrics['start_time'])
            end = datetime.fromisoformat(metrics['end_time'])
            duration = (end - start).total_seconds()
            lines.append(f"总执行时间: {duration:.2f} 秒")
        
        lines.append("\n" + "-" * 60)
        lines.append("  Token 消耗统计")
        lines.append("-" * 60)
        tokens = metrics['total_tokens']
        lines.append(f"输入 Tokens: {tokens['input']:,}")
        lines.append(f"输出 Tokens: {tokens['output']:,}")
        lines.append(f"总计 Tokens: {tokens['total']:,}")
        
        lines.append("\n按 Agent 分布:")
        for agent, usage in metrics['tokens_by_agent'].items():
            lines.append(f"  {agent}:")
            lines.append(f"    输入: {usage['input']:,}")
            lines.append(f"    输出: {usage['output']:,}")
            lines.append(f"    总计: {usage['total']:,}")
        
        lines.append("\n" + "-" * 60)
        lines.append("  Agent 执行统计")
        lines.append("-" * 60)
        for agent, count in metrics['agent_executions'].items():
            timings = metrics['agent_timings'].get(agent, [])
            avg_time = sum(timings) / len(timings) if timings else 0
            lines.append(f"{agent}:")
            lines.append(f"  执行次数: {count}")
            lines.append(f"  平均耗时: {avg_time:.2f} 秒")
        
        lines.append("\n" + "-" * 60)
        lines.append("  任务统计")
        lines.append("-" * 60)
        tasks = metrics['tasks']
        lines.append(f"计划任务数: {tasks['planned']}")
        lines.append(f"完成任务数: {tasks['completed']}")
        lines.append(f"失败任务数: {tasks['failed']}")
        lines.append(f"重试次数: {tasks['retried']}")
        if tasks['planned'] > 0:
            success_rate = (tasks['completed'] / tasks['planned']) * 100
            lines.append(f"成功率: {success_rate:.1f}%")
        
        lines.append("\n" + "-" * 60)
        lines.append("  验证统计")
        lines.append("-" * 60)
        validation = metrics['validation_results']
        lines.append(f"验证通过: {validation['passed']}")
        lines.append(f"验证失败: {validation['failed']}")
        total_validations = validation['passed'] + validation['failed']
        if total_validations > 0:
            pass_rate = (validation['passed'] / total_validations) * 100
            lines.append(f"通过率: {pass_rate:.1f}%")
        
        if metrics['errors']:
            lines.append("\n" + "-" * 60)
            lines.append("  错误记录")
            lines.append("-" * 60)
            for i, error in enumerate(metrics['errors'], 1):
                lines.append(f"{i}. [{error['agent']}] {error['timestamp']}")
                lines.append(f"   {error['error']}")
        
        lines.append("\n" + "=" * 60)
        
        return "\n".join(lines)
    
    @staticmethod
    def save_report(output_dir: str, task_id: str):
        """保存报告到文件"""
        import os
        tracker = MetricsTracker()
        metrics = tracker.get_metrics()
        
        os.makedirs(output_dir, exist_ok=True)
        
        # 保存 JSON 格式
        json_path = os.path.join(output_dir, f"metrics_{task_id}.json")
        with open(json_path, 'w', encoding='utf-8') as f:
            json.dump(metrics, f, indent=2, ensure_ascii=False)
        
        # 保存文本报告
        txt_path = os.path.join(output_dir, f"report_{task_id}.txt")
        with open(txt_path, 'w', encoding='utf-8') as f:
            f.write(MetricsReporter.generate_summary(metrics))
        
        print(f"指标报告已保存:")
        print(f"  JSON: {json_path}")
        print(f"  文本: {txt_path}")
        
        return json_path, txt_path
```

```python
# principia_ai/metrics/middleware.py
from typing import Any, List
from langchain.schema import BaseMessage
from langchain.callbacks.base import BaseCallbackHandler
from .tracker import MetricsTracker

class MetricsCallbackHandler(BaseCallbackHandler):
    """LangChain 回调处理器，自动追踪 LLM 调用"""
    
    def __init__(self, agent_name: str):
        self.agent_name = agent_name
        self.tracker = MetricsTracker()
        self.start_time = None
    
    def on_llm_start(self, serialized: dict, prompts: List[str], **kwargs):
        """LLM 开始时调用"""
        import time
        self.start_time = time.time()
    
    def on_llm_end(self, response, **kwargs):
        """LLM 结束时调用"""
        import time
        duration = time.time() - self.start_time if self.start_time else 0
        
        # 提取 token 信息
        if hasattr(response, 'llm_output') and response.llm_output:
            token_usage = response.llm_output.get('token_usage', {})
            self.tracker.record_llm_call(
                agent_name=self.agent_name,
                input_tokens=token_usage.get('prompt_tokens', 0),
                output_tokens=token_usage.get('completion_tokens', 0),
                model=token_usage.get('model_name', 'unknown'),
                duration=duration
            )
    
    def on_llm_error(self, error: Exception, **kwargs):
        """LLM 出错时调用"""
        self.tracker.record_error(self.agent_name, str(error))

class MetricsLLMWrapper:
    """LLM 包装器，自动添加追踪"""
    
    def __init__(self, llm, agent_name: str):
        self.llm = llm
        self.agent_name = agent_name
        self.tracker = MetricsTracker()
    
    def invoke(self, messages, **kwargs):
        """包装 invoke 方法"""
        import time
        start_time = time.time()
        
        # 添加回调
        callbacks = kwargs.get('callbacks', [])
        callbacks.append(MetricsCallbackHandler(self.agent_name))
        kwargs['callbacks'] = callbacks
        
        try:
            response = self.llm.invoke(messages, **kwargs)
            return response
        except Exception as e:
            self.tracker.record_error(self.agent_name, str(e))
            raise
    
    def __getattr__(self, name):
        """代理其他属性到原始 LLM"""
        return getattr(self.llm, name)
```

```python
# principia_ai/metrics/graph_hooks.py
from typing import Dict, Any
from principia_ai.graph.graph_state import GraphState
from .tracker import MetricsTracker

class GraphMetricsHooks:
    """LangGraph 执行钩子"""
    
    def __init__(self):
        self.tracker = MetricsTracker()
    
    def before_node(self, node_name: str, state: GraphState):
        """节点执行前"""
        self.tracker.start_agent(node_name)
        
        # 特殊处理
        if node_name == "orchestrator" and state.get('plan'):
            # 记录计划任务数
            self.tracker.metrics['tasks']['planned'] = len(state['plan'])
    
    def after_node(self, node_name: str, state: GraphState):
        """节点执行后"""
        self.tracker.end_agent(node_name)
        
        # 记录任务状态变化
        if state.get('completed_tasks'):
            self.tracker.metrics['tasks']['completed'] = len(state['completed_tasks'])
        
        if state.get('validation_status') == 'passed':
            self.tracker.record_validation(True)
        elif state.get('validation_status') == 'failed':
            self.tracker.record_validation(False)
            self.tracker.record_task_event('retried')
```

```python
# principia_ai/metrics/storage.py
import sqlite3
import json
from datetime import datetime
from typing import Dict, Any, List
from contextlib import contextmanager

class MetricsStorage:
    """持久化存储指标数据"""
    
    def __init__(self, db_path: str = "./metrics_db/metrics.db"):
        self.db_path = db_path
        self._init_db()
    
    @contextmanager
    def _get_connection(self):
        conn = sqlite3.connect(self.db_path)
        conn.row_factory = sqlite3.Row
        try:
            yield conn
            conn.commit()
        except Exception:
            conn.rollback()
            raise
        finally:
            conn.close()
    
    def _init_db(self):
        """初始化数据库表"""
        with self._get_connection() as conn:
            cursor = conn.cursor()
            
            # 任务表
            cursor.execute('''
                CREATE TABLE IF NOT EXISTS tasks (
                    task_id TEXT PRIMARY KEY,
                    user_request TEXT,
                    start_time TEXT,
                    end_time TEXT,
                    duration_seconds REAL,
                    status TEXT,
                    total_input_tokens INTEGER,
                    total_output_tokens INTEGER,
                    total_tokens INTEGER,
                    created_at TEXT DEFAULT CURRENT_TIMESTAMP
                )
            ''')
            
            # LLM 调用表
            cursor.execute('''
                CREATE TABLE IF NOT EXISTS llm_calls (
                    id INTEGER PRIMARY KEY AUTOINCREMENT,
                    task_id TEXT,
                    agent_name TEXT,
                    timestamp TEXT,
                    input_tokens INTEGER,
                    output_tokens INTEGER,
                    total_tokens INTEGER,
                    model TEXT,
                    duration REAL,
                    FOREIGN KEY (task_id) REFERENCES tasks(task_id)
                )
            ''')
            
            # Agent 执行表
            cursor.execute('''
                CREATE TABLE IF NOT EXISTS agent_executions (
                    id INTEGER PRIMARY KEY AUTOINCREMENT,
                    task_id TEXT,
                    agent_name TEXT,
                    start_time TEXT,
                    end_time TEXT,
                    duration REAL,
                    status TEXT,
                    FOREIGN KEY (task_id) REFERENCES tasks(task_id)
                )
            ''')
            
            # 任务事件表
            cursor.execute('''
                CREATE TABLE IF NOT EXISTS task_events (
                    id INTEGER PRIMARY KEY AUTOINCREMENT,
                    task_id TEXT,
                    event_type TEXT,
                    event_data TEXT,
                    timestamp TEXT,
                    FOREIGN KEY (task_id) REFERENCES tasks(task_id)
                )
            ''')
            
            # 验证结果表
            cursor.execute('''
                CREATE TABLE IF NOT EXISTS validations (
                    id INTEGER PRIMARY KEY AUTOINCREMENT,
                    task_id TEXT,
                    agent_name TEXT,
                    passed BOOLEAN,
                    notes TEXT,
                    timestamp TEXT,
                    FOREIGN KEY (task_id) REFERENCES tasks(task_id)
                )
            ''')
            
            # 错误记录表
            cursor.execute('''
                CREATE TABLE IF NOT EXISTS errors (
                    id INTEGER PRIMARY KEY AUTOINCREMENT,
                    task_id TEXT,
                    agent_name TEXT,
                    error_message TEXT,
                    error_type TEXT,
                    timestamp TEXT,
                    FOREIGN KEY (task_id) REFERENCES tasks(task_id)
                )
            ''')
    
    def save_task(self, task_data: Dict[str, Any]):
        """保存任务记录"""
        with self._get_connection() as conn:
            cursor = conn.cursor()
            cursor.execute('''
                INSERT OR REPLACE INTO tasks 
                (task_id, user_request, start_time, end_time, duration_seconds,
                 status, total_input_tokens, total_output_tokens, total_tokens)
                VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?)
            ''', (
                task_data['task_id'],
                task_data.get('user_request'),
                task_data.get('start_time'),
                task_data.get('end_time'),
                task_data.get('duration_seconds'),
                task_data.get('status'),
                task_data.get('total_input_tokens', 0),
                task_data.get('total_output_tokens', 0),
                task_data.get('total_tokens', 0)
            ))
    
    def save_llm_call(self, task_id: str, call_data: Dict[str, Any]):
        """保存 LLM 调用记录"""
        with self._get_connection() as conn:
            cursor = conn.cursor()
            cursor.execute('''
                INSERT INTO llm_calls 
                (task_id, agent_name, timestamp, input_tokens, output_tokens,
                 total_tokens, model, duration)
                VALUES (?, ?, ?, ?, ?, ?, ?, ?)
            ''', (
                task_id,
                call_data['agent'],
                call_data['timestamp'],
                call_data['input_tokens'],
                call_data['output_tokens'],
                call_data['total_tokens'],
                call_data['model'],
                call_data.get('duration')
            ))
    
    def get_task_summary(self, task_id: str) -> Dict[str, Any]:
        """获取任务摘要"""
        with self._get_connection() as conn:
            cursor = conn.cursor()
            
            # 任务基本信息
            cursor.execute('SELECT * FROM tasks WHERE task_id = ?', (task_id,))
            task = dict(cursor.fetchone())
            
            # LLM 调用统计
            cursor.execute('''
                SELECT agent_name, 
                       COUNT(*) as call_count,
                       SUM(input_tokens) as total_input,
                       SUM(output_tokens) as total_output,
                       AVG(duration) as avg_duration
                FROM llm_calls 
                WHERE task_id = ?
                GROUP BY agent_name
            ''', (task_id,))
            task['llm_stats'] = [dict(row) for row in cursor.fetchall()]
            
            return task
    
    def get_all_tasks(self, limit: int = 100) -> List[Dict[str, Any]]:
        """获取所有任务列表"""
        with self._get_connection() as conn:
            cursor = conn.cursor()
            cursor.execute('''
                SELECT task_id, user_request, start_time, end_time,
                       duration_seconds, status, total_tokens
                FROM tasks
                ORDER BY created_at DESC
                LIMIT ?
            ''', (limit,))
            return [dict(row) for row in cursor.fetchall()]
```

```python
# principia_ai/metrics/collector.py
from .tracker import MetricsTracker
from .storage import MetricsStorage
from datetime import datetime

class MetricsCollector:
    """扩展的指标收集器，支持持久化"""
    
    def __init__(self, db_path: str = "./metrics_db/metrics.db"):
        self.tracker = MetricsTracker()
        self.storage = MetricsStorage(db_path)
    
    def start_task(self, task_id: str, user_request: str):
        """开始任务并记录到数据库"""
        self.tracker.start_task(task_id)
        self.tracker.metrics['user_request'] = user_request
        
        self.storage.save_task({
            'task_id': task_id,
            'user_request': user_request,
            'start_time': datetime.now().isoformat(),
            'status': 'running'
        })
    
    def end_task(self, status: str = 'completed'):
        """结束任务并保存到数据库"""
        self.tracker.end_task()
        
        metrics = self.tracker.get_metrics()
        
        # 计算持续时间
        if metrics['start_time'] and metrics['end_time']:
            start = datetime.fromisoformat(metrics['start_time'])
            end = datetime.fromisoformat(metrics['end_time'])
            duration = (end - start).total_seconds()
        else:
            duration = 0
        
        # 保存任务摘要
        self.storage.save_task({
            'task_id': metrics['task_id'],
            'user_request': metrics.get('user_request'),
            'start_time': metrics['start_time'],
            'end_time': metrics['end_time'],
            'duration_seconds': duration,
            'status': status,
            'total_input_tokens': metrics['total_tokens']['input'],
            'total_output_tokens': metrics['total_tokens']['output'],
            'total_tokens': metrics['total_tokens']['total']
        })
        
        # 保存所有 LLM 调用
        for call in metrics['llm_calls']:
            self.storage.save_llm_call(metrics['task_id'], call)
```
